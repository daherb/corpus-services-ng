55  @Override
56  public Report function(CorpusData cd, Boolean fix) // check whether there's any illegal apostrophes '
57  throws SAXException, IOException, ParserConfigurationException, URISyntaxException, JDOMException, TransformerException, XPathExpressionException {
58  Report stats = new Report(); // create a new report
59  doc = TypeConverter.String2JdomDocument(cd.toSaveableString()); // read the file as a doc
60  CorpusIO cio = new CorpusIO();
61 
62  //temporary stopgap to prevent working on files with multiple speakers
63  Boolean stopgap = false;
64  XPath xpathStopgap = XPath.newInstance("//tier[@category='ts'");
65  List stopList = xpathStopgap.selectNodes(doc);
66  if (stopList.size() > 1) {
67  stopgap = true;
68  }
69 
70  if(!stopgap) {
71  ArrayList<String> timeline = new ArrayList<>();
72  ArrayList<String> originalEnd = new ArrayList<>();
73  Map<String,ArrayList<String>> sentenceMap = new HashMap<>();
74  Map<String,ArrayList<String>> timeMap = new HashMap<>();
75 
76  //first we collect all the timeline elements in a list
77  XPath XTimeline = XPath.newInstance("//tli");
78  List allTimestamps = XTimeline.selectNodes(doc);
79  for (int tl = 0; tl < allTimestamps.size(); tl ++) {
80  Object to = allTimestamps.get(tl);
81  if (to instanceof Element) {
82  Element te = (Element) to;
83  String tID = te.getAttributeValue("id");
84  //aborting the check if the file is not normalized
85  if (!tID.equals("T" + tl)) {
86  System.out.println("EXB is not normalized; aborting");
87  System.exit(0);
88  }
89  timeline.add(tID);
90  }
91  }
92 
93  //put the sentence from the ref tier and the associated timeline IDs into a HashMap
94  XPath xSentences = XPath.newInstance("//tier[@category='ref']/event");
95  List allSentences = xSentences.selectNodes(doc);
96  for (int sent = 0; sent < allSentences.size(); sent++) {
97  Object o = allSentences.get(sent);
98  if (o instanceof Element) {
99  Element e = (Element) o;
100  ArrayList<String> sentenceList = new ArrayList<>();
101  String start = e.getAttributeValue("start");
102  String end = e.getAttributeValue("end");
103  originalEnd.add(end);
104  Integer eStart = timeline.indexOf(start);
105  Integer eEnd = timeline.indexOf(end);
106  for (int it = eStart; it < eEnd; it++) {
107  sentenceList.add(timeline.get(it));
108  }
109  sentenceMap.put(e.getText(), sentenceList);
110  }
111  }
112 
113  //put the sentence from the ref tier and the associated time codes IDs into a HashMap
114  for (String sentenceId : sentenceMap.keySet()) {
115  List ids = sentenceMap.get(sentenceId);
116  ArrayList<String> timeList = new ArrayList<>();
117  for (int i = 0; i < ids.size(); i++) {
118  String xpathGetTli = "//tli[@id='" + ids.get(i) + "']";
119  XPath getTli = XPath.newInstance(xpathGetTli);
120  Object gotTliObj = getTli.selectSingleNode(doc);
121  if (gotTliObj instanceof Element) {
122  Element gotTliEle = (Element) gotTliObj;
123  String timeStamp = gotTliEle.getAttributeValue("time");
124  if (timeStamp != null) {
125  timeList.add(timeStamp);
126  }
127  }
128  }
129  timeMap.put(sentenceId, timeList);
130  }
131 
132  //reformat HashMaps to TreeMaps for the preservation of ordering
133  //TODO: check what happens to the ordering if we initiate TreeMaps instead of HashMaps in the first place
134  TreeMap<String,ArrayList<String>> sentenceTree = new TreeMap<>(sentenceMap);
135  TreeMap<String,ArrayList<String>> timeTree = new TreeMap<>(timeMap);
136 
137  //create a new timeline and populate it with events for pauses
138  Element newTimeline = new Element("common-timeline");
139  TreeMap<String, String> endTimes = new TreeMap<>();
140  int timelineSize = allTimestamps.size();
141  for (String sentenceRef : sentenceTree.keySet()) {
142  List refs = sentenceTree.get(sentenceRef);
143  List times = timeTree.get(sentenceRef);
144  String startID = "";
145  for (int ref = 0; ref < refs.size(); ref++) {
146  Element newTli = new Element("tli");
147  newTli.setAttribute("id", refs.get(ref).toString());
148  if (ref == 0) {
149  startID = refs.get(ref).toString();
150  newTli.setAttribute("time", times.get(ref).toString());
151  }
152  newTimeline.addContent(newTli);
153  if (ref == (refs.size() - 1) && refs.size() > 1) {
154  //only create pauses for sentences which have two or more words
155  String pauseId = "T" + timelineSize;
156  timelineSize ++;
157  endTimes.put(startID, pauseId);
158  Element pauseTli = new Element("tli");
159  pauseTli.setAttribute("id", pauseId);
160  if (times.size() > 1 ) {
161  pauseTli.setAttribute("time", times.get(times.size() - 1).toString());
162  }
163  newTimeline.addContent(pauseTli);
164  }
165  }
166  }
167 
168  //replace the old timeline with a new one
169  Object body = XPath.newInstance("//basic-body").selectSingleNode(doc);
170  if (body instanceof Element) {
171  Element eBody = (Element) body;
172  Object oldTime = eBody.getChild("common-timeline");
173  if (oldTime instanceof Element) {
174  Element eOldTime = (Element) oldTime;
175  eOldTime.detach();
176  }
177  eBody.addContent(0, newTimeline);
178  }
179 
180  //set the original "end" attributes to the pauses created above
181  //TODO: what would happen if the file is normalized before this step?
182  for (String startTli : endTimes.keySet()) {
183  String xpathEvents = "//event[@start='" + startTli + "']";
184  XPath events = XPath.newInstance(xpathEvents);
185  List relevantEvents = events.selectNodes(doc);
186  for (int event = 0; event < relevantEvents.size(); event++) {
187  Object evo = relevantEvents.get(event);
188  if (evo instanceof Element) {
189  Element eve = (Element) evo;
190  String ending = eve.getAttributeValue("end");
191  if (originalEnd.contains(ending)) {
192  eve.setAttribute("end", endTimes.get(startTli));
193  String xpathWords = "//event[@end='" + ending + "']";
194  XPath words = XPath.newInstance(xpathWords);
195  List relevantWords = words.selectNodes(doc);
196  for (int word = 0; word < relevantWords.size(); word ++) {
197  Object wo = relevantWords.get(word);
198  if (wo instanceof Element) {
199  Element we = (Element) wo;
200  we.setAttribute("end", endTimes.get(startTli));
201  }
202  }
203  }
204  }
205  }
206  }
207 
208  //write out the file
209  if (fix) {
210  cd.updateUnformattedString(TypeConverter.JdomDocument2String(doc));
211  cio.write(cd, cd.getURL());
212  stats.addFix(function, cd, "Corrected the timeline");
213  }
214  }
215  return stats;
216  }
