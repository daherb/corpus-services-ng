53  throws SAXException, IOException, ParserConfigurationException, URISyntaxException, JDOMException, TransformerException, XPathExpressionException {
54  Report stats = new Report(); // create a new report
55  flexFile = cd.toSaveableString();
56  doc = TypeConverter.String2JdomDocument(flexFile); // read the file as a doc
57  Pattern bracketsPattern = Pattern.compile("[«»]");
58  Pattern slashPattern = Pattern.compile("^…\\/$");
59  context = XPath.newInstance(xpathContext);
60  List allContextInstances = context.selectNodes(doc);
61  CorpusIO cio = new CorpusIO();
62  String s = "";
63  if (!allContextInstances.isEmpty()) {
64  for (int i = 0; i < allContextInstances.size(); i++) {
65  Object o = allContextInstances.get(i);
66  if (o instanceof Element) {
67  Element e = (Element) o;
68  s = e.getText();
69  if (bracketsPattern.matcher(s).find()) { // if file contains the RegEx then issue warning
70  badPunctuation = true;
71  stats.addCritical(function, cd, "Remove «» brackets in :" + s);
72  }
73  if (slashPattern.matcher(s).find()) {
74  badPunctuation = true;
75  stats.addCritical(function, cd, "Add a slash before " + s);
76  }
77  }
78  }
79  if (badPunctuation && fix) {
80  flexFile = flexFile.replaceAll("«", "“");
81  flexFile = flexFile.replaceAll("»", "”");
83  //testRegEx = flexFile.matches("(?:[^\\/])…\\/");
84  flexFile = flexFile.replaceAll("(?<=\\>)…\\/", "\\/…\\/");
85  //flexFile = flexFile.replaceAll("…\\/", "\\/…\\/");
86  //flexFile = flexFile.replaceAll("\\/\\/…", "\\/…");
87  cd.updateUnformattedString(flexFile);
88  cio.write(cd, cd.getURL());
89  stats.addFix(function, cd, "Corrected punctuation");
90  }
91  if (!badPunctuation) {
92  stats.addCorrect(function, cd, "CorpusData file does not contain bad punctuation");
93  }
94  } else {
95  stats.addCorrect(function, cd, "CorpusData file does not contain any transcription");
96  }
97  return stats; // return the report with warnings
98  }
